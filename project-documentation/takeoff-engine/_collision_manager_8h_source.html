94 {
95 TE_ASSERT_MSG(inFraction >= 0.0f && inFraction <= 1.0f,

107 bool RayPoint(TEfloat* t, Vec& hitPoint) const
108 {
109 TEfloat time = (glm::dot(hitPoint, mDirection) - dot(mOrigin, mDirection)) / dot(mDirection, mDirection);
110 if (t) *t = time; return time >= 0.0f && time <= 1.0f;
