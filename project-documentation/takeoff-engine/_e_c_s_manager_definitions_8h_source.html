120 TE_STATIC_ASSERT((std::is_base_of<IComponent, ComponentType>::value, "ComponentType must inherit IComponent"));
121 static TEtypeguid typeguid = TETYPEGUID_NULL;

123 // check if this func ran before. If yes, means registered, don't run again.

139 TE_DEBUG("ECS: Registered component %s", typeid(ComponentType).name());

190 void ECSManager::ComponentContainer<ComponentType>::_AllocateComponent(ECSIndex entityIndex, TEguid existingGuid) {
191 TE_ASSERT_MSG(entityIndex < m_ECSIndexToCompIndex.size(), "invalid entityIndex: %u", entityIndex);
192 TE_ASSERT_MSG(m_CompArray.size() < ECS_MAX_ENTITIES,
193 "idk how but there are more components than entites now... ComponentType: %s [%llu]",
194 typeid(ComponentType).name(),
195 GUIDCreator::getTypeGUID<ComponentType>);
196 if (m_ECSIndexToCompIndex[entityIndex] != ECS_COMPID_NULL) {
197 TE_WARN("AddComponent: This entity %s [%llu] already has this component %s[%llu].\nIgnoring this add component.",
198 ECSManager::GetInstanceHandle().m_EntityStorage[entityIndex].m_EntityName.c_str(),
199 ECSManager::GetInstanceHandle().m_EntityStorage[entityIndex].m_EntityGUID,
200 typeid(ComponentType).name(),
201 GUIDCreator::getTypeGUID<ComponentType>);
202 return;
203 }
204
205 m_ECSIndexToCompIndex[entityIndex] = (ComponentID)m_CompArray.size();
206 m_CompArray.emplace_back();
207 m_CompArray.back().m_BindedEntityIndex = entityIndex;
208 if (existingGuid != TEGUID_NULL) m_CompArray.back().SetGUID(existingGuid);
209 m_CompArrayVersion++;
210
211 TE_TRACE("Added component %s [%llu] to entity %s[%llu].\nCompArray usage: %llu/%u",
212 typeid(ComponentType).name(),
213 GUIDCreator::getTypeGUID<ComponentType>,
214 ECSManager::GetInstanceHandle().m_EntityStorage[entityIndex].m_EntityName.c_str(),
215 ECSManager::GetInstanceHandle().m_EntityStorage[entityIndex].m_EntityGUID,
216 m_CompArray.size(), ECS_MAX_ENTITIES);

219 template <typename ComponentType>
220 void ECSManager::ComponentContainer<ComponentType>::_DeallocateComponent(ECSIndex entityIndex) {
221 TE_TRACE("Deallocating component %s [%llu] from entity %s[%llu].\nCompArray usage (before): %llu/%u",
222 typeid(ComponentType).name(),
223 GUIDCreator::getTypeGUID<ComponentType>,
224 ECSManager::GetInstanceHandle().m_EntityStorage[entityIndex].m_EntityName.c_str(),
225 ECSManager::GetInstanceHandle().m_EntityStorage[entityIndex].m_EntityGUID,
226 m_CompArray.size(), ECS_MAX_ENTITIES);
227 TE_ASSERT_MSG(entityIndex < m_ECSIndexToCompIndex.size(), "invalid entityIndex: %u", entityIndex);
228 if (m_ECSIndexToCompIndex[entityIndex] == ECS_COMPID_NULL) {
229 TE_WARN("_DeallocateComponent: This entity %s [%llu] does not have this component %s[%llu].\nIgnoring this remove component.",
230 ECSManager::GetInstanceHandle().m_EntityStorage[entityIndex].m_EntityName.c_str(),
231 ECSManager::GetInstanceHandle().m_EntityStorage[entityIndex].m_EntityGUID,
232 typeid(ComponentType).name(),
233 GUIDCreator::getTypeGUID<ComponentType>);
234 return;
235 }
236
237 m_CompArray[m_ECSIndexToCompIndex[entityIndex]] = m_CompArray.back(); // move last guy into entityIndex slot
238 m_ECSIndexToCompIndex[m_CompArray.back().m_BindedEntityIndex] = m_ECSIndexToCompIndex[entityIndex]; // last guy now uses entityIndex slot
239 m_ECSIndexToCompIndex[entityIndex] = ECS_COMPID_NULL; // entityIndex now has no component
240 m_CompArray.pop_back(); // remove last component
241 m_CompArrayVersion++;
242
243 TE_TRACE("CompArray usage (after): %llu/%u",
244 m_CompArray.size(), ECS_MAX_ENTITIES);
245 }

279 TE_ASSERT_MSG(fromentity < m_ECSIndexToCompIndex.size(), "invalid fromentity entityIndex: %u", fromentity);
280 TE_ASSERT_MSG(toEntity < m_ECSIndexToCompIndex.size(), "invalid toEntity entityIndex: %u", toEntity);

320 std::vector<ComponentRef<ComponentType>> ECSManager::GetComponentArray() {
321 TE_STATIC_ASSERT((std::is_base_of<IComponent, ComponentType>::value, "ComponentType must inherit IComponent"));
322 TE_ASSERT_MSG(CheckComponentRegistered<ComponentType>(), "Component %s not registered in ECS!", typeid(ComponentType).name());
323 static std::vector<ComponentRef<ComponentType>> ret{};
324 static decltype(_GetComponentContainer<ComponentType>().m_CompArrayVersion) lastVersion = 0;
325 if (lastVersion == _GetComponentContainer<ComponentType>().m_CompArrayVersion) return ret;
326
327 const std::vector<ComponentType>& comparr = _GetComponentContainer<ComponentType>().m_CompArray;
328 ret.clear();
329 ret.reserve(comparr.size());
330 for (const ComponentType& comp : comparr) {
331 ret.emplace_back(comp.m_BindedEntityIndex);
332 }
333 lastVersion = _GetComponentContainer<ComponentType>().m_CompArrayVersion;
334 return ret;
335 }

345 template <typename ComponentType>
346 ComponentRef<ComponentType> ECSManager::FindComponent(TEguid componentGuid) {
347 static decltype(_GetComponentContainer<ComponentType>().m_CompArrayVersion) lastVersion = 0;
348 static std::unordered_map<TEguid, ECSIndex> map;
349 if (lastVersion == _GetComponentContainer<ComponentType>().m_CompArrayVersion) {
350 if (map.contains(componentGuid)) return map[componentGuid];
351 TE_WARN("FindComponent: component type %s with GUID %llu not found!", typeid(ComponentType).name(), componentGuid);
352 return ComponentRef<ComponentType>();
353 }
354 // rebuild map if outdated
355 auto& compContainer = _GetComponentContainer<ComponentType>();
356 std::vector<ComponentType>& comparr = compContainer.m_CompArray;
357 map.clear();
358 for (ComponentType& comp : comparr) {
359 map[comp.GetGUID()] = comp.m_BindedEntityIndex;
360 }
361 lastVersion = compContainer.m_CompArrayVersion;
362
363 if (map.contains(componentGuid)) return map[componentGuid];
364 TE_WARN("FindComponent: component type %s with GUID %llu not found!", typeid(ComponentType).name(), componentGuid);
365 return ComponentRef<ComponentType>();
366 }
