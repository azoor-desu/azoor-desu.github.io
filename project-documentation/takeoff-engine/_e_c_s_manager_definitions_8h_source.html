118 template <typename ComponentType>
119 void ECSManager::RegisterComponent() {
120 TE_STATIC_ASSERT((std::is_base_of<IComponent, ComponentType>::value, "ComponentType must inherit IComponent"));
121 static TEtypeguid typeguid = TETYPEGUID_NULL;
122
123 // check if this func ran before. If yes, means registered, don't run again.
124 if (typeguid != TETYPEGUID_NULL) {
125 TE_WARN("ECS: This component %s is already registered, ignoring.", typeid(ComponentType).name());
126 return;
127 }
128 typeguid = GUIDCreator::getTypeGUID<ComponentType>;
129
130 // add this component to back of m_ComponentStorage
131 m_ComponentStorage.reserve(m_ComponentStorage.size() + 1); // force a new if needed
132 // TODO: If mem manager, add flag here to mark that this next new call is for ComponentContainer.
133 m_ComponentStorage.emplace_back(
134 std::unique_ptr<IComponentContainer>((IComponentContainer*)(new ComponentContainer<ComponentType>))
135 );
136 // TODO: If mem manager, add end flag here
137
138 GUIDCreator::SetTypeGuidToName(typeguid, rttr::type::get<ComponentType>().get_name().data()); // register the typeguid to name conversion
139 TE_DEBUG("ECS: Registered component %s", typeid(ComponentType).name());
140 }

189 template <typename ComponentType>
190 void ECSManager::ComponentContainer<ComponentType>::_AllocateComponent(ECSIndex entityIndex, TEguid existingGuid) {
191 TE_ASSERT_MSG(entityIndex < m_ECSIndexToCompIndex.size(), "invalid entityIndex: %u", entityIndex);
192 TE_ASSERT_MSG(m_CompArray.size() < ECS_MAX_ENTITIES,
193 "idk how but there are more components than entites now... ComponentType: %s [%llu]",
194 typeid(ComponentType).name(),
195 GUIDCreator::getTypeGUID<ComponentType>);
196 if (m_ECSIndexToCompIndex[entityIndex] != ECS_COMPID_NULL) {
197 TE_WARN("AddComponent: This entity %s [%llu] already has this component %s[%llu].\nIgnoring this add component.",
198 ECSManager::GetInstanceHandle().m_EntityStorage[entityIndex].m_EntityName.c_str(),
199 ECSManager::GetInstanceHandle().m_EntityStorage[entityIndex].m_EntityGUID,
200 typeid(ComponentType).name(),
201 GUIDCreator::getTypeGUID<ComponentType>);
202 return;
203 }
204
205 m_ECSIndexToCompIndex[entityIndex] = (ComponentID)m_CompArray.size();
206 m_CompArray.emplace_back();
207 m_CompArray.back().m_BindedEntityIndex = entityIndex;
208 if (existingGuid != TEGUID_NULL) m_CompArray.back().SetGUID(existingGuid);
209 m_CompArrayVersion++;
210
211 m_CompArray.back().AllocateData();
212
213 TE_TRACE("Added component %s [%llu] to entity %s[%llu].\nCompArray usage: %llu/%u",
214 typeid(ComponentType).name(),
215 GUIDCreator::getTypeGUID<ComponentType>,
216 ECSManager::GetInstanceHandle().m_EntityStorage[entityIndex].m_EntityName.c_str(),
217 ECSManager::GetInstanceHandle().m_EntityStorage[entityIndex].m_EntityGUID,
218 m_CompArray.size(), ECS_MAX_ENTITIES);
219 }

221 template <typename ComponentType>
222 void ECSManager::ComponentContainer<ComponentType>::_DeallocateComponent(ECSIndex entityIndex) {
223 TE_TRACE("Deallocating component %s [%llu] from entity %s[%llu].\nCompArray usage (before): %llu/%u",
224 typeid(ComponentType).name(),
225 GUIDCreator::getTypeGUID<ComponentType>,
226 ECSManager::GetInstanceHandle().m_EntityStorage[entityIndex].m_EntityName.c_str(),
227 ECSManager::GetInstanceHandle().m_EntityStorage[entityIndex].m_EntityGUID,
228 m_CompArray.size(), ECS_MAX_ENTITIES);
229 TE_ASSERT_MSG(entityIndex < m_ECSIndexToCompIndex.size(), "invalid entityIndex: %u", entityIndex);
230 if (m_ECSIndexToCompIndex[entityIndex] == ECS_COMPID_NULL) {
231 TE_WARN("_DeallocateComponent: This entity %s [%llu] does not have this component %s[%llu].\nIgnoring this remove component.",
232 ECSManager::GetInstanceHandle().m_EntityStorage[entityIndex].m_EntityName.c_str(),
233 ECSManager::GetInstanceHandle().m_EntityStorage[entityIndex].m_EntityGUID,
234 typeid(ComponentType).name(),
235 GUIDCreator::getTypeGUID<ComponentType>);
236 return;
237 }
238
239
240 // update the sparse array to point to the correct guys, and swap data values.
241 ECSIndex& sparseArrayDel = m_ECSIndexToCompIndex[entityIndex];
242 ECSIndex& sparseArraySwap = m_ECSIndexToCompIndex[m_CompArray.back().m_BindedEntityIndex];
243 ComponentType& compDel = m_CompArray[sparseArrayDel];
244 ComponentType& compSwap = m_CompArray.back();
245
246 compDel.DeallocateData();
247
248 sparseArraySwap = sparseArrayDel;
249 sparseArrayDel = ECS_COMPID_NULL;
250 std::swap(compDel, compSwap);
251 m_CompArray.pop_back(); // remove last component
252 m_CompArrayVersion++;
253
254 TE_TRACE("CompArray usage (after): %llu/%u",
255 m_CompArray.size(), ECS_MAX_ENTITIES);
256 }

288 template <typename ComponentType>
289 void ECSManager::ComponentContainer<ComponentType>::_SoftCopyComponentData(ECSIndex fromentity, ECSIndex toEntity) {
290 TE_ASSERT_MSG(fromentity < m_ECSIndexToCompIndex.size(), "invalid fromentity entityIndex: %u", fromentity);
291 TE_ASSERT_MSG(toEntity < m_ECSIndexToCompIndex.size(), "invalid toEntity entityIndex: %u", toEntity);
292 TE_ASSERT(m_ECSIndexToCompIndex[fromentity] < m_CompArray.size());
293 TE_ASSERT(m_ECSIndexToCompIndex[toEntity] < m_CompArray.size());
294 // copy everything except for the component's binded entity GUID, its own GUID and soft active state.
295 // specifically, transform/scene graph need to avoid changing parent too.
296 // Maybe have to ask ppl to create their own template specializations for edge cases like this too.
297 // possible to declare a template specialization in the .h file of the transform class.
298
299 TEguid guid = m_CompArray[m_ECSIndexToCompIndex[toEntity]].GetGUID();
300 ECSIndex bindID = m_CompArray[m_ECSIndexToCompIndex[toEntity]].m_BindedEntityIndex;
301
302 m_CompArray[m_ECSIndexToCompIndex[toEntity]] = m_CompArray[m_ECSIndexToCompIndex[fromentity]];
303
304 m_CompArray[m_ECSIndexToCompIndex[toEntity]].SetGUID(guid);
305 m_CompArray[m_ECSIndexToCompIndex[toEntity]].m_BindedEntityIndex = bindID;
306 }

313 template <typename ComponentType>
314 ECSManager::ComponentContainer<ComponentType>& ECSManager::_GetComponentContainer() {
315 constexpr TEtypeguid typeguid = GUIDCreator::getTypeGUID<ComponentType>;
316 static ComponentContainer<ComponentType>* ptr = nullptr; // cache for fast lookup
317 if (ptr != nullptr) return *ptr;
318
319 for (auto& containerPtr : m_ComponentStorage) {
320 if (typeguid == containerPtr->m_ComponentTypeGuid) {
321 ptr = static_cast<ComponentContainer<ComponentType>*>(containerPtr.get());
322 return *ptr;
323 }
324 }
325 TE_ASSERT_MSG(false, "unable to find component type %s, is this component registered?", typeid(ComponentType).name());
326 throw std::bad_typeid();
327 }

330 template <typename ComponentType>
331 std::vector<ComponentRef<ComponentType>> ECSManager::GetComponentArray() {
332 TE_STATIC_ASSERT((std::is_base_of<IComponent, ComponentType>::value, "ComponentType must inherit IComponent"));
333 TE_ASSERT_MSG(CheckComponentRegistered<ComponentType>(), "Component %s not registered in ECS!", typeid(ComponentType).name());
334 static std::vector<ComponentRef<ComponentType>> ret{};
335 static decltype(_GetComponentContainer<ComponentType>().m_CompArrayVersion) lastVersion = 0;
336 if (lastVersion == _GetComponentContainer<ComponentType>().m_CompArrayVersion) return ret;
337
338 const std::vector<ComponentType>& comparr = _GetComponentContainer<ComponentType>().m_CompArray;
339 ret.clear();
340 ret.reserve(comparr.size());
341 for (const ComponentType& comp : comparr) {
342 ret.emplace_back(comp.m_BindedEntityIndex);
343 }
344 lastVersion = _GetComponentContainer<ComponentType>().m_CompArrayVersion;
345 return ret;
346 }

356 template <typename ComponentType>
357 ComponentRef<ComponentType> ECSManager::FindComponent(TEguid componentGuid) {
358 static decltype(_GetComponentContainer<ComponentType>().m_CompArrayVersion) lastVersion = 0;
359 static std::unordered_map<TEguid, ECSIndex> map;
360 if (lastVersion == _GetComponentContainer<ComponentType>().m_CompArrayVersion) {
361 if (map.contains(componentGuid)) return map[componentGuid];
362 TE_WARN("FindComponent: component type %s with GUID %llu not found!", typeid(ComponentType).name(), componentGuid);
363 return ComponentRef<ComponentType>();
364 }
365 // rebuild map if outdated
366 auto& compContainer = _GetComponentContainer<ComponentType>();
367 std::vector<ComponentType>& comparr = compContainer.m_CompArray;
368 map.clear();
369 for (ComponentType& comp : comparr) {
370 map[comp.GetGUID()] = comp.m_BindedEntityIndex;
371 }
372 lastVersion = compContainer.m_CompArrayVersion;
373
374 if (map.contains(componentGuid)) return map[componentGuid];
375 TE_WARN("FindComponent: component type %s with GUID %llu not found!", typeid(ComponentType).name(), componentGuid);
376 return ComponentRef<ComponentType>();
377 }
