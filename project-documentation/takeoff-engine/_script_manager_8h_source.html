36 // A threadsafe-queue.
37 // https://stackoverflow.com/questions/15278343/c11-thread-safe-queue
38 template <class T>
39 class SafeQueue {
40 public:
41 SafeQueue(void): m_Queue(), m_Mutex(), m_CondVar() {}
42
43 ~SafeQueue(void) {}
44
45 // Add an element to the queue.
46 void enqueue(T t) {
47 std::lock_guard<std::mutex> lock(m_Mutex);
48 m_Queue.push(t);
49 m_CondVar.notify_one();
50 }
51
52 // Get the "front"-element.
53 // If the queue is empty, wait till a element is avaiable.
54 T dequeue() {
55 std::unique_lock<std::mutex> lock(m_Mutex);
56 while (m_Queue.empty()) {
57 // release lock as long as the wait and ream_Queueuire it afterwards.
58 m_CondVar.wait(lock);
59 }
60 T val = m_Queue.front();
61 m_Queue.pop();
62 return val;
63 }
64
65 bool empty() {
66 return m_Queue.empty();
67 }
68
69 private:
70 std::queue<T> m_Queue;
71 mutable std::mutex m_Mutex;
72 std::condition_variable m_CondVar;
73 };

269 static DataType ScriptManager::GetMonoFieldValue(_MonoObject* objectInstance, const MonoFieldContainer& fieldContainer) {
270
271 if (fieldContainer.GetIsStatic()) {
272 //TE_ERROR("Static mono fields not supported yet!!!!!"); // TODO static mono fields
273 return DataType();
274 }
275
276 if (CheckMonoFieldTypeIncompatible<DataType>(fieldContainer.GetMonoFieldType())) {
277 TE_ERROR("Unable to do get data, types are incompatible!");
278 return DataType();
279 }
280
281 if (fieldContainer.GetMonoFieldType() == MONO_TYPE_ARRAY ||
282 fieldContainer.GetMonoFieldType() == MONO_TYPE_SZARRAY) {
283 TE_ERROR("GET ARRAYS UNIMPLEMENTED!");
284 return DataType();
285 }
286 else if (fieldContainer.GetMonoFieldType() == MONO_TYPE_CLASS) {
287 TE_ERROR("GET CLASSES UNIMPLEMENTED!");
288 return DataType();
289 }
290 else {
291 DataType ret{};
292 _MonoGetFieldValue(objectInstance, fieldContainer.m_Field, &ret);
293 return ret;
294 }
295 }
