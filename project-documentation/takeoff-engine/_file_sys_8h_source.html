29 static inline std::string FileSys_AppendFolderPath(const T& ...args) {
30 std::string ret;
31 std::string_view views[]{args...};
32 std::string::size_type full_size = 0;
33
34 // calculate full path length.
35 // add an additional slash for each guy.
36 for (const auto& sub_view : views) {
37 full_size += sub_view.size() + 1;
38 }
39
40 // reserve enough space in string and append file paths to it.
41 ret.reserve(full_size);
42 ret = views[0]; // first view will be skipped.
43 for (const auto& sub_view : views | std::views::drop(1)) { // ranged base for loop, but skip first element.
44 if (sub_view.empty()) continue;
45 if (ret.back() != '\\'
46 && ret.back() != '/'
47 && sub_view.back() != '\\'
48 && sub_view.back() != '/') ret.append("\\");
49 ret.append(sub_view);
50 }
51 return ret;
52 }
