496 T RemoveFileExtension(T const& filename)

498 typename T::size_type const p(filename.find_last_of('.'));

503 template <class T>
504 void EditorHelper::Inspector_FormattedCenteredButton(const TEchar* label, const TEchar* buttonText, T* state, TEfloat alignment, INSPECTORSTATE INSTATE, const TEchar* typeDragDrop)
505 {
506 //Name
507 ImGui::PushID(label);
508 ImGui::Columns(2);
509 ImGui::SetColumnWidth(0, 100.f);
510 ImGui::Text(label);
511 ImGui::NextColumn();
512
513 ImGuiStyle& style = ImGui::GetStyle();
514
515 float size = ImGui::CalcTextSize(label).x + style.FramePadding.x * 2.0f;
516 float avail = ImGui::GetContentRegionAvail().x;
517
518 float off = (avail - size) * alignment;
519 if (off > 0.0f)
520 ImGui::SetCursorPosX(ImGui::GetCursorPosX() + off);
521 switch (INSTATE) {
522 case DRAGDROP:
523 if (ImGui::Button(buttonText)) {
524 *state = !(*state);
525 }
526 if (ImGui::BeginDragDropTarget()) {
527 if (const ImGuiPayload* payload = ImGui::AcceptDragDropPayload(typeDragDrop)) {
528 const wchar_t* path = (const wchar_t*)payload->Data;
529 const std::filesystem::path fName = path;
530 TEstring filePath = fName.string();
531 *state = filePath;
532 }
533 ImGui::EndDragDropTarget();
534 }
535 break;
536 case DRAGDROPOPEN:
537 if (ImGui::Button(buttonText)) {
538 TEstring filePath = OpenFile("");
539 *state = filePath;
540 }
541 if (ImGui::BeginDragDropTarget()) {
542 if (const ImGuiPayload* payload = ImGui::AcceptDragDropPayload(typeDragDrop)) {
543 const wchar_t* path = (const wchar_t*)payload->Data;
544 const std::filesystem::path fName = path;
545 TEstring filePath = fName.string();
546 *state = filePath;
547 }
548 ImGui::EndDragDropTarget();
549 }
550 break;
551 case OPENONLY:
552 if (ImGui::Button(buttonText)) {
553 TEstring filePath = OpenFile("");
554 *state = filePath;
555 }
556 break;
557 case DEFAULT:
558 if (ImGui::Button(buttonText)) {
559 *state = !(*state);
560 }
561 break;
562 }
563
564 ImGui::Columns(1);
565 ImGui::PopID();
566 }
