425 template <class T>
426 void EditorHelper::Inspector_FormattedCenteredButton(const TEchar* label, const TEchar* buttonText, T* state, TEfloat alignment, INSPECTORSTATE INSTATE, const TEchar* typeDragDrop)
427 {
428 //Name
429 ImGui::PushID(label);
430 ImGui::Columns(2);
431 ImGui::SetColumnWidth(0, 100.f);
432 ImGui::Text(label);
433 ImGui::NextColumn();
434
435 ImGuiStyle& style = ImGui::GetStyle();
436
437 float size = ImGui::CalcTextSize(label).x + style.FramePadding.x * 2.0f;
438 float avail = ImGui::GetContentRegionAvail().x;
439
440 float off = (avail - size) * alignment;
441 if (off > 0.0f)
442 ImGui::SetCursorPosX(ImGui::GetCursorPosX() + off);
443 switch (INSTATE) {
444 case DRAGDROP:
445 if (ImGui::Button(buttonText)) {
446 *state = !(*state);
447 }
448 if (ImGui::BeginDragDropTarget()) {
449 if (const ImGuiPayload* payload = ImGui::AcceptDragDropPayload(typeDragDrop)) {
450 const wchar_t* path = (const wchar_t*)payload->Data;
451 const std::filesystem::path fName = path;
452 TEstring filePath = fName.string();
453 *state = filePath;
454 }
455 ImGui::EndDragDropTarget();
456 }
457 break;
458 case DRAGDROPOPEN:
459 if (ImGui::Button(buttonText)) {
460 TEstring filePath = OpenFile("");
461 *state = filePath;
462 }
463 if (ImGui::BeginDragDropTarget()) {
464 if (const ImGuiPayload* payload = ImGui::AcceptDragDropPayload(typeDragDrop)) {
465 const wchar_t* path = (const wchar_t*)payload->Data;
466 const std::filesystem::path fName = path;
467 TEstring filePath = fName.string();
468 *state = filePath;
469 }
470 ImGui::EndDragDropTarget();
471 }
472 break;
473 case OPENONLY:
474 if (ImGui::Button(buttonText)) {
475 TEstring filePath = OpenFile("");
476 *state = filePath;
477 }
478 break;
479 case DEFAULT:
480 if (ImGui::Button(buttonText)) {
481 *state = !(*state);
482 }
483 break;
484 }
485 }
