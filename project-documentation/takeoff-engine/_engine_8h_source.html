317 ManagerType& Engine::GetManager() {
318 constexpr TEtypeguid managerType = GUIDCreator::getTypeGUID<ManagerType>;
319 ManagerType* ret = nullptr;
320 for (int i = 0; i < m_Managers.size(); i++) {
321 if (m_Managers[i].first == managerType) {
322 ret = static_cast<ManagerType*>(m_Managers[i].second);
323 break;
324 }
325 }
326 TE_ASSERT_MSG(ret != nullptr,
327 "GetManagerRef(): Unable to find manager class %s in the engine. Is it not added to the engine?",
328 typeid(ManagerType).name());

342 template <class ManagerType>
343 void Engine::_AddManager(bool runOnlyDuringSimulation) {
344 constexpr TEtypeguid typeGUID = GUIDCreator::getTypeGUID<ManagerType>;
345#ifdef _DEBUG
346 // check if manager already exists
347 for (int i = 0; i < m_Managers.size(); i++) {
348 if (m_Managers[i].first == typeGUID)
349 TE_ASSERT_MSG(false,
350 "Multiple mangers of same type %s not allowed in a single engine.\n",
351 typeid(ManagerType).name()
352 );
353 }
354#endif // _DEBUG
355
356 ManagerType* newManager = new ManagerType;
357 newManager->m_RunOnlyDuringSimulation = runOnlyDuringSimulation;
358 m_Managers.emplace_back(typeGUID, newManager);
359
360 TE_DEBUG("New manager added: [%s]. GUID [%llu]. typeid: [%llu]\n",
361 newManager->DEBUG_GetName(),
362 newManager->GetGUID(),
363 newManager->GetTypeGUID()
364 );
365 GUIDCreator::SetTypeGuidToName(typeGUID, typeid(ManagerType).name());
366 }

369 // Removes a Manager system from the engine.
370 template <typename ManagerType>
371 void Engine::_RemoveManager() {
372 constexpr TEtypeguid managerType = GUIDCreator::getTypeGUID<ManagerType>;
373 size_t toDelete = -1;
374 for (size_t i = 0; i < m_Managers.size(); i++) {
375 if (m_Managers[i].first == managerType) {
376 m_Managers[i].second->ExitManager(); // safely exit manager before deleting
377 delete static_cast<ManagerType*>(m_Managers[i].second);
378 toDelete = i;
379 break;
380 }
381 }
382 TE_ASSERT_MSG(toDelete != -1, "_RemoveManager: Manager %s not found. Probably not added or is already removed.", typeid(ManagerType).name());
383 m_Managers.erase(m_Managers.begin() + toDelete);
384 }
