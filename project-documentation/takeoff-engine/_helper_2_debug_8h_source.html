124 // Logging system:
125 // Able to capture ALL console output, and store the messages internally.
126 // Able to toggle levels to save lesser messages, for Release mode.
127 // Able to redirect messages to console AND internal console logs.
128 // Able to allow user to quickly send debug messages in code.
129 // Able to write to Log file.
130
131 // the GUID of this Logging System.
132 // Use this value to subscribe to LogEvent, and recieve messages.
133 extern const TEguid debuggerGUID;
134
135 enum class LogLevel {
136 LOG_FATAL, // for things that go very very wrong and program has to stop immediately.
137 LOG_ERROR, // for things that go wrong, potentially causing game to crash. Program can still run enough to log the happenings.
138 LOG_WARN, // for things that went wrong but are not game-breaking.
139 LOG_INFO, // for general information and state of engine e.g. manager init, exit etc.
140 LOG_DEBUG, // Like info, but with more system-specific details (manager does this then that then that)
141 LOG_TRACE, // for very detailed details (count of stuff, IDs, etc) about very specific systems.
142 LOG_PERF // ONLY FOR PERFORMANCE TRACKER!!!!
143 };
