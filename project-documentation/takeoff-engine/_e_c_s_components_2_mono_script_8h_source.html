126 template<typename DataType>
127 DataType MonoScript::GetDataFieldValue(MonoFieldID fieldID) {
128 TE_ASSERT_MSG(Engine::GetInstanceHandle()->GetManager<ScriptManager>().GetMonoClasses().contains(m_MonoClassType), "MonoScript's class type does not exist in loaded assembly");
129 TE_ASSERT_MSG(Engine::GetInstanceHandle()->GetManager<ScriptManager>().GetMonoClasses().at(m_MonoClassType).m_ClassFields.contains(fieldID), "MonoScript's loaded assembly does not contain this field.");
130 // normal editor mode: access the overriden values using m_Values. If no override, get from monoobject.
131 if (!Engine::GetInstanceHandle()->IsEngineSimulating()) {
132 if (m_Values.contains(fieldID)) return std::any_cast<DataType>(m_Values.at(fieldID));
133 else {
134 const MonoFieldContainer& fieldContainer = Engine::GetInstanceHandle()->GetManager<ScriptManager>().GetMonoClasses().at(m_MonoClassType).m_ClassFields.at(fieldID);
135 return ScriptManager::GetMonoFieldValue<DataType>(m_ObjInstance, fieldContainer);
136 }
137 }
138 // simulation mode: assume the "overriden" values have been pushed into each monoobject, only grab values from
139 // monoobjects.
140 else {
141 const MonoFieldContainer& fieldContainer = Engine::GetInstanceHandle()->GetManager<ScriptManager>().GetMonoClasses().at(m_MonoClassType).m_ClassFields.at(fieldID);
142 return ScriptManager::GetMonoFieldValue<DataType>(m_ObjInstance, fieldContainer);
143 }

153 void MonoScript::SetDataFieldValue(MonoFieldID fieldID, const DataType& value, bool overwriteMonoObjectVal) {
154 TE_ASSERT_MSG(Engine::GetInstanceHandle()->GetManager<ScriptManager>().GetMonoClasses().contains(m_MonoClassType), "MonoScript's class type does not exist in loaded assembly");
155 TE_ASSERT_MSG(Engine::GetInstanceHandle()->GetManager<ScriptManager>().GetMonoClasses().at(m_MonoClassType).m_ClassFields.contains(fieldID), "Loaded assembly does not contain this field.");
156
157 const MonoFieldContainer& fieldContainer = Engine::GetInstanceHandle()->GetManager<ScriptManager>().GetMonoClasses().at(m_MonoClassType).m_ClassFields.at(fieldID);
158
159 // normal editor mode: always set value to m_Values, then set into monoobject based on overwriteMonoObjectVal bool.
160 if (!Engine::GetInstanceHandle()->IsEngineSimulating()) {
161 m_Values[fieldID] = value;
162 if (overwriteMonoObjectVal) {
163 ScriptManager::SetMonoFieldValue(m_ObjInstance, fieldContainer, value);
164 }
165 }
166 // simulation mode: always set into monoobject, ignoring overwriteMonoObjectVal bool.
167 else {
168 ScriptManager::SetMonoFieldValue(m_ObjInstance, fieldContainer, value);
169 }
