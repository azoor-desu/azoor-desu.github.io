26// How to use:
27// 1. Create a struct/class for Messages, inheriting from TE::MessageBase.
28// 2. Put your desired message contents inside this class. This will be used to deliver messages. Leave the class empty if needed.
29// 3. Create a MsgSender instance. Bind it to a GUID of an existing object that has a GUID.
30// 4. Somewhere else, create MsgReciever. Subscribe to sender using that GUID and another function created somewhere acting as a callback.
31// 5. When sender wants to send messages, do sender.SendToSubscribers<MessageClass>(messageifany);
32// Code example:
33/*
34// ============ in a shared .h file ============
35// Create message classes/structs. Sender and all recievers need to use this.
36
37struct EmptyMessage: public TE::MessageBase {};
38struct OnCollisionMessage: public TE::MessageBase {
39 std::string msg;
40};
41
42// ============ in FIRST .cpp file ============
43// Create a sender
44
45TE::MsgSender sender(SOMEGUID);
46
47
48// ============ in SECOND .cpp file ============
49// create functions and a reciever.
50// WARNING: All reciever functions [OnXXX()] need to have NO params or const MsgType& params!
51// ONLY DO OnXXX(const MsgType&) or OnXXX(MsgType const&) or OnXXX()
52// don't do OnXXX(MsgType&) or OnXXX(MsgType*) or OnXXX(MsgType) or OnXXX(const MsgType)
53
54class FunctionsInAClass {
55public:
56 void OnEmptyMsgMemberFunc() { printf("OnEmptyMsgMemberFunc\n"); }
57 void OnMsgMemberFunc(const OnCollisionMessage& msg) { printf("OnMsgMemberFunc: %s\n", msg.msg.c_str()); }
58};
59void OnEmptyMsgGlobalFunc() { printf("OnEmptyMsgGlobalFunc\n"); }
60void OnMsgGlobalFunc(const OnCollisionMessage& msg) { printf("OnMsgGlobalFunc: %s\n", msg.msg.c_str()); }
61
62void init() {
63 TE::MsgReceiver rcv; // create reciever
64 FunctionsInAClass objInstance; // create example object to demo subscribing to Member Functions
65
66 // subscribe to MsgSender
67 rcv.SubscribeToMsg<FunctionsInAClass, const OnCollisionMessage&, &FunctionsInAClass::OnMsgMemberFunc>(SOMEGUID, &objInstance); // subscribing to MEMBER FUNCTION, where callback takes in a message
68 rcv.SubscribeToMsg<FunctionsInAClass, const EmptyMessage&, &FunctionsInAClass::OnEmptyMsgMemberFunc>(SOMEGUID, &objInstance); // subscribing to MEMBER FUNCTION, where callback has no parameter
69 rcv.SubscribeToMsg<const OnCollisionMessage&, &OnMsgGlobalFunc>(SOMEGUID); // subscribing to GLOBAL FUNCTION, where callback takes in a message
70 rcv.SubscribeToMsg<const EmptyMessage&, &OnEmptyMsgGlobalFunc>(SOMEGUID); // subscribing to GLOBAL FUNCTION, where callback has no parameter
71
72 // reciever now has 4 subscriptions to sender:
73 // 2 subscriptions for EmptyMessage, 2 subscriptions for OnCollisionMessage
74 // when sender sends out EmptyMessage for example, ALL of the functions subscribed using EmptyMessage will trigger.
75}
76
77// ============ back to FIRST .cpp file ============
78// sender is ready to send out events.
79
80void init() {
81 OnCollisionMessage msg; // create message to send. Not required for empty messages.
82 msg.msg = "Hello!"; // assign data to message
83
84 sender.SendToSubscribers<EmptyMessage>(); // prints out "OnEmptyMsgMemberFunc" and "OnEmptyMsgGlobalFunc"
85 sender.SendToSubscribers<OnCollisionMessage>(msg); // prints out "OnMsgMemberFunc: Hello!" and "OnMsgGlobalFunc: Hello!"
86}

209 // A bunch of BinderFuncs will be generated with templates, Each one will store this data:
210 // 1. The raw callback function pointer to be called
211 // 2. The correct conversion types to convert from a generic type to a type that the raw callback can use
212 // 3. The way to call these functions. the way to call normal func ptrs is slightly different from member func ptrs.
213 // The pointer to each uniquely generated BinderFunc() should be saved and used later.
214 // Since the resultant BinderFunc() all have the same signature, it is now possible to store all these differnt guys together in a single container.
215 template <typename ObjectType, typename FuncParam, void(ObjectType::* MemberFunc)(FuncParam)>
216 static void BinderFunc(void* instance, const MessageBase* msgObject) {
217 TE_ASSERT_MSG(instance != nullptr, "this subscription expects an object instance of type %s but none was given!", typeid(ObjectType).name());
218 TE_VALIDATE_MSG(msgObject != nullptr, "this subscription expects a message but none was given!");
219 (reinterpret_cast<ObjectType*>(instance)->*MemberFunc)(static_cast<const FuncParam&>(*msgObject));
220 }

222 static void BinderFunc(void* instance, const MessageBase* msgObject) {
223 TE_ASSERT_MSG(instance != nullptr, "this subscription expects an object instance of type %s but none was given!", typeid(ObjectType).name());
224 TE_VALIDATE_MSG(msgObject == nullptr, "this subscription does not expect a message but one was given");
225 UNUSED_PARAM(msgObject);
226 (reinterpret_cast<ObjectType*>(instance)->*MemberFunc)();
227 }
228 template <typename FuncParam, void(*FuncPtr)(FuncParam)>
229 static void BinderFunc(void* instance, const MessageBase* msgObject) {
230 TE_VALIDATE_MSG(instance == nullptr, "this subscription does not expect a instance but one was given!");
231 TE_VALIDATE_MSG(msgObject != nullptr, "this subscription expects a message but none was given!");
232 UNUSED_PARAM(instance);
233 (*FuncPtr)(static_cast<const FuncParam&>(*msgObject));
234 }
235 template <void(*FuncPtr)(void)>
236 static void BinderFunc(void* instance, const MessageBase* msgObject) {
237 TE_VALIDATE_MSG(instance == nullptr, "this subscription does not expect a instance but one was given!");
238 TE_VALIDATE_MSG(msgObject == nullptr, "this subscription does not expect a message but one was given");
239 UNUSED_PARAM(instance);
240 UNUSED_PARAM(msgObject);
241 (*FuncPtr)();
242 }

487 template <typename MsgType>
488 void MsgSender::SendMessageDirect(TEguid recieverGUID, const MsgType& message) {
489 constexpr MsgTypeGUID msgTypeGUID = GUIDCreator::getTypeGUID<MsgType>;
490
491 TE_ASSERT_MSG(m_SenderSubscribers.contains(msgTypeGUID),
492 "[MessageSystem] SendMessageDirect: Sender does not have this msgType %s registered.",
493 typeid(MsgType).name()
494 );
495
496 TE_ASSERT_MSG(!m_SenderSubscribers[msgTypeGUID].empty(),
497 "[MessageSystem] SendMessageDirect: Sender's msgType %s contains nobody.",
498 typeid(MsgType).name()
499 );
500
501 // find guys to send to.
502 // loose matching: Send message to anyone that is of this message type AND is in this reciever.
503 auto& subscribers = m_SenderSubscribers[msgTypeGUID];
504
505 for (SenderSubscription& subscription : subscribers) {
506 if (subscription.m_ReceiverGUID == recieverGUID) {
507 if (!subscription.m_Enabled) continue;
508 subscription.m_FuncPtr(subscription.m_ObjectInstance, &message);
509 }
510 }
511 }

567 void MsgReceiver::SubscribeToMsg(TEguid senderBindedGUID, void* objectInstance) {
568 constexpr MsgTypeGUID msgTypeGUID = GUIDCreator::getTypeGUID<FuncParam>;
569 // generate unique identification for this subscription.
570 // BinderFunc generates a function with that template, and that function pointer gets saved.
571 static MsgSender::BinderFuncPtr func = &MsgSender::BinderFunc<ObjectType, FuncParam, MemberFunc>;
572 MsgSender::SubscriptionID subscriptionID = MsgSender::_CreateSubscriptionID(func, objectInstance);
573
574 // Debug: If senderBindedGUID exists AND subscriptionID matches, subscription already exists.
575 TE_ASSERT_MSG(!m_RcvSubscriptions.contains(senderBindedGUID)
576 || std::find(
577 m_RcvSubscriptions[senderBindedGUID].begin(),
578 m_RcvSubscriptions[senderBindedGUID].end(),
579 subscriptionID) == m_RcvSubscriptions[senderBindedGUID].end(),
580 "[MessageSystem] SubscribeToMsg: This receiver is already subscribed to GUID %llu with the MsgType %s. Ignoring this subscription request.",
581 senderBindedGUID,
582 typeid(FuncParam).name()
583 );
584
585 // Debug: Sender does net exist in All Senders map!
586 TE_ASSERT_MSG(MsgSender::_GetAllSendersMap().contains(senderBindedGUID),
587 "[MessageSystem] SubscribeToMsg: Sender with GUID %llu does not exist in All Senders map.",
588 senderBindedGUID
589 );
590
591 // save to internal subscription list for future lookup
592 m_RcvSubscriptions[senderBindedGUID].emplace_back(msgTypeGUID, subscriptionID);
593
594 // tell sender to add this subscription to their list too.
595 MsgSender::_GetAllSendersMap()[senderBindedGUID]->_SubscribeToMsg(msgTypeGUID,
596 m_ReceiverGUID,
597 func,
598 objectInstance
599 );
600#ifdef _DEBUG
601 // Get the most recent subscription and set the DEBUG_RequiresMsg to true/false
602 MsgSender::_GetAllSendersMap()[senderBindedGUID]->m_SenderSubscribers[msgTypeGUID].back().DEBUG_RequiresMsg = true;
603#endif // _DEBUG
604
605 }
